Step 5 — Days Calculator Function
This calculates how many days are left between today and February 1, 2027 (GATE exam date — change year if needed).
Open your existing MainActivity.kt file
Add this function anywhere inside the class, below onCreate:
MainActivity.kt — add this function
fun getDaysRemaining(): Int {
    val gateDate = Calendar.getInstance().apply {
        set(2027, Calendar.FEBRUARY, 1)  // ← Change year if needed
        set(Calendar.HOUR_OF_DAY, 0)
        set(Calendar.MINUTE, 0)
        set(Calendar.SECOND, 0)
    }
    val today = Calendar.getInstance()
    val diffMs = gateDate.timeInMillis - today.timeInMillis
    return maxOf(0, (diffMs / (1000 * 60 * 60 * 24)).toInt())
}

Step 6 — Create the AI Image Generator
This file talks to Together AI, sends a prompt, and downloads the generated image as a Bitmap (image object in Android).
Right-click your package folder → New → Kotlin File/Class → name it ImageGenerator
Paste the full code below. Replace YOUR_API_KEY_HERE with your actual key from Step 2:
ImageGenerator.kt — new file
import android.graphics.Bitmap
import android.graphics.BitmapFactory
import android.util.Base64
import okhttp3.*
import okhttp3.MediaType.Companion.toMediaType
import okhttp3.RequestBody.Companion.toRequestBody
import org.json.JSONObject

object ImageGenerator {

    private const val API_KEY = "YOUR_API_KEY_HERE"  // ← Paste your key here
    private const val API_URL = "https://api.together.xyz/v1/images/generations"
    private const val MODEL = "black-forest-labs/FLUX.1-schnell-free"

    fun generate(days: Int): Bitmap? {
        val prompt = """
            Motivational dark study background, deep space cosmic theme,
            glowing engineering diagrams, circuit patterns, soft bokeh lights,
            dramatic blue and gold tones, no text anywhere, ultra HD,
            portrait phone wallpaper, cinematic, inspiring atmosphere
        """.trimIndent()

        val jsonBody = JSONObject().apply {
            put("model", MODEL)
            put("prompt", prompt)
            put("width", 1080)
            put("height", 1920)
            put("steps", 4)
            put("response_format", "b64_json")
        }

        val request = Request.Builder()
            .url(API_URL)
            .addHeader("Authorization", "Bearer $API_KEY")
            .post(jsonBody.toString()
                .toRequestBody("application/json".toMediaType()))
            .build()

        return try {
            val response = OkHttpClient().newCall(request).execute()
            val body = response.body?.string() ?: return null
            val b64 = JSONObject(body)
                .getJSONArray("data")
                .getJSONObject(0)
                .getString("b64_json")
            val bytes = Base64.decode(b64, Base64.DEFAULT)
            BitmapFactory.decodeByteArray(bytes, 0, bytes.size)
        } catch (e: Exception) {
            e.printStackTrace()
            null
        }
    }
}

import android.graphics.*

object WallpaperComposer {

    fun compose(base: Bitmap, days: Int): Bitmap {
        val result = base.copy(Bitmap.Config.ARGB_8888, true)
        val canvas = Canvas(result)
        val w = result.width.toFloat()
        val h = result.height.toFloat()

        // Dark overlay in center zone
        Paint().apply {
            color = Color.argb(160, 0, 0, 0)
        }.also { canvas.drawRect(0f, h*0.50f, w, h*0.80f, it) }

        // Big day number
        Paint().apply {
            color = Color.WHITE
            textSize = 240f
            typeface = Typeface.create(Typeface.DEFAULT, Typeface.BOLD)
            textAlign = Paint.Align.CENTER
            setShadowLayer(30f, 0f, 0f, Color.parseColor("#00D4FF"))
        }.also { canvas.drawText("$days", w/2f, h*0.66f, it) }

        // "DAYS REMAINING" label
        Paint().apply {
            color = Color.parseColor("#00D4FF")
            textSize = 52f
            typeface = Typeface.create(Typeface.DEFAULT, Typeface.BOLD)
            textAlign = Paint.Align.CENTER
            letterSpacing = 0.15f
        }.also { canvas.drawText("DAYS REMAINING", w/2f, h*0.73f, it) }

        // "TO GATE EXAM" label
        Paint().apply {
            color = Color.argb(200, 255, 255, 255)
            textSize = 36f
            textAlign = Paint.Align.CENTER
            letterSpacing = 0.2f
        }.also { canvas.drawText("TO GATE EXAM", w/2f, h*0.77f, it) }

        return result
    }
}

Step 8 — Create the Background Worker
This is the most important file. It runs every day automatically, calls the image generator, adds text, and sets the wallpaper — all in the background while you sleep.
Create new file → name it WallpaperWorker
WallpaperWorker.kt — new file
import android.app.WallpaperManager
import android.content.Context
import androidx.work.*
import java.util.Calendar

class WallpaperWorker(
    private val ctx: Context,
    params: WorkerParameters
) : Worker(ctx, params) {

    override fun doWork(): Result {
        return try {
            // 1. Calculate days
            val days = getDaysRemaining()

            // 2. Generate AI image
            val bitmap = ImageGenerator.generate(days)
                ?: return Result.retry()  // Retry if generation fails

            // 3. Add countdown text overlay
            val finalBitmap = WallpaperComposer.compose(bitmap, days)

            // 4. Set as wallpaper
            val manager = WallpaperManager.getInstance(ctx)
            manager.setBitmap(finalBitmap)

            Result.success()
        } catch (e: Exception) {
            e.printStackTrace()
            Result.retry()
        }
    }

    private fun getDaysRemaining(): Int {
        val gateDate = Calendar.getInstance().apply {
            set(2027, Calendar.FEBRUARY, 1)
        }
        val diff = gateDate.timeInMillis - System.currentTimeMillis()
        return maxOf(0, (diff / (1000L * 60 * 60 * 24)).toInt())
    }
}

Step 9 — Update MainActivity to Schedule the Worker
This starts the daily job when you open the app for the first time. After that, it runs on its own every day at midnight.
MainActivity.kt — replace your onCreate with this
import androidx.appcompat.app.AppCompatActivity
import android.os.Bundle
import android.widget.Button
import android.widget.TextView
import androidx.work.*
import java.util.Calendar
import java.util.concurrent.TimeUnit

class MainActivity : AppCompatActivity() {

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        // Show days remaining on screen
        val days = getDaysRemaining()
        findViewById<TextView>(R.id.tvDays).apply {
            text = "$days days to GATE"
        }

        // Button: run immediately to test
        findViewById<Button>(R.id.btnSetNow).setOnClickListener {
            val oneTimeWork = OneTimeWorkRequestBuilder<WallpaperWorker>().build()
            WorkManager.getInstance(this).enqueue(oneTimeWork)
        }

        // Schedule daily at midnight
        scheduleDailyWallpaper()
    }

    private fun scheduleDailyWallpaper() {
        val delay = getDelayUntilMidnight()
        val dailyWork = PeriodicWorkRequestBuilder<WallpaperWorker>(
            1, TimeUnit.DAYS
        ).setInitialDelay(delay, TimeUnit.MILLISECONDS).build()

        WorkManager.getInstance(this).enqueueUniquePeriodicWork(
            "gate_wallpaper",
            ExistingPeriodicWorkPolicy.KEEP,
            dailyWork
        )
    }

    private fun getDelayUntilMidnight(): Long {
        val midnight = Calendar.getInstance().apply {
            add(Calendar.DAY_OF_YEAR, 1)
            set(Calendar.HOUR_OF_DAY, 0)
            set(Calendar.MINUTE, 0)
            set(Calendar.SECOND, 0)
        }
        return midnight.timeInMillis - System.currentTimeMillis()
    }

    private fun getDaysRemaining(): Int {
        val gateDate = Calendar.getInstance().apply {
            set(2027, Calendar.FEBRUARY, 1)
        }
        val diff = gateDate.timeInMillis - System.currentTimeMillis()
        return maxOf(0, (diff / (1000L * 60 * 60 * 24)).toInt())
    }
}

Step 11 — Enable Developer Mode on Your Phone
This lets Android Studio install apps directly on your phone via USB.
Go to Settings → About Phone
Tap Build Number 7 times repeatedly — you'll see "Developer mode enabled"
Go back to Settings → now you'll see Developer Options
Inside Developer Options → turn on USB Debugging
Connect your phone to PC via USB cable → tap Allow on the popup
▶️
Step 12 — Build & Run the App
This compiles all your code and installs it on your phone.
In Android Studio, at the top you'll see a device dropdown — select your phone name
Click the Green Play ▶ button (or press Shift+F10)
Wait 1–2 minutes for it to build and install
Your phone will open the app automatically
You'll see the days remaining displayed on screen
Click "Set Wallpaper Now (Test)" to test it immediately
Wait 30–60 seconds (AI image is generating) → check your wallpaper!
✅
After the first test works, the daily schedule is already set. You don't need to open the app again. Every midnight it will auto-generate a new wallpaper and update it for you.
